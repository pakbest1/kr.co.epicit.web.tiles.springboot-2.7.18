DROP TABLE IF EXISTS TNCM_ARTICLE;
CREATE TABLE TNCM_ARTICLE_GROUP  (
    ARTICLE_ID     VARCHAR(32) NOT NULL,
    ARTICLE_TITLE  VARCHAR(1000) NOT NULL,
    DELETE_AT      VARCHAR(1) NOT NULL DEFAULT 'N',
    CREATED_BY     VARCHAR(100),
    CREATED_AT     TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UPDATED_BY     VARCHAR(100),
    UPDATED_AT     TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (ARTICLE_ID)
);

DROP TABLE IF EXISTS TNCM_ARTICLE_CONTENT;
CREATE TABLE TNCM_ARTICLE_CONTENT (
    ARTICLE_ID     VARCHAR(32) NOT NULL,
    CONTENT_ID     VARCHAR(32) NOT NULL,
    CONTENT_TYPE   VARCHAR(32) NOT NULL,
    CONTENT_TITLE  VARCHAR(1000) NOT NULL,
    CONTENT        CLOB NOT NULL,
    DELETE_AT      VARCHAR(1) NOT NULL DEFAULT 'N',
    CREATED_BY     VARCHAR(100),
    CREATED_AT     TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UPDATED_BY     VARCHAR(100),
    UPDATED_AT     TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    VIEW_COUNT     INT DEFAULT 0,
    PRIMARY KEY (ARTICLE_ID, CONTENT_ID)
);
